// Hidden passive that upgrades the borrowed sandking_epicenter when the hero holds a Scepter.

// possible without mutating real Sand King. This is the sanctioned tiny-Lua numeric scepter:
//   - -scepter_cooldown_reduction_pct% Epicenter cooldown (per-ability gated modifier property)
//   - +scepter_bonus_damage_pct% Epicenter damage (ON_TAKEDAMAGE amp, gated + re-entrancy guarded)

		"AbilityType"			"ABILITY_TYPE_BASIC"
		"AbilityTextureName"	"sandking_epicenter"
		"MaxLevel"				"1"
		"IsBreakable"			"0"

		"AbilityValues"
		{
			"scepter_cooldown_reduction_pct"	"40"	// -% Epicenter cooldown while holding Scepter
			"scepter_bonus_damage_pct"			"50"	// +% Epicenter damage while holding Scepter
		}
	}
}
